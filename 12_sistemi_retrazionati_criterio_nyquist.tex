\section{Sistemi retroazionati: criterio di Nyquist e margini di stabilità}

\subsection{Buona connessione}

\begin{definition}
  Il sistema retroazionato è detto ben connesso se:
  \begin{align}
    \lim_{s\to\infty} 1 + L(s) \neq 0
  \end{align}
\end{definition}
